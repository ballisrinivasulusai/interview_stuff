"""

AirPot_2 Unit test application.

These APIs can be reused for testing the AirPot_2 which
can be called from the some other file.

UID : 2akq

"""

import time
from tinkerforge.bricklet_silent_stepper_v2 import BrickletSilentStepperV2
from TinkerForge.tforge_con import create_connection
from TinkerForge.AirPot_2_config_2akq import AirPot_2_UID, AP2_MOTOR_CONFIG, AP2_UPPER_LIMIT, AP2_LOWER_LIMIT

def ap2_initialize_stepper(ipcon):
    """
    Initializes the Silent Stepper V2 Bricklet for AirPot-1.

    Returns:
        BrickletSilentStepperV2: The initialized stepper instance.
        IPConnection: The Tinkerforge IP connection instance.
    """
    stepper = BrickletSilentStepperV2(AirPot_2_UID, ipcon)
    ap2_configure_stepper(stepper)
    return stepper

def ap2_configure_stepper(stepper):
    """
    Configures the stepper motor with predefined parameters.

    Args:
        stepper (BrickletSilentStepperV2): The stepper motor instance.
    """
    stepper.set_step_configuration(BrickletSilentStepperV2.STEP_RESOLUTION_64, 0)  # 1/256 Microstepping
    stepper.set_motor_current(AP2_MOTOR_CONFIG["run_current"])
    stepper.set_max_velocity(AP2_MOTOR_CONFIG["max_velocity"])
    stepper.set_speed_ramping(AP2_MOTOR_CONFIG["acceleration"], AP2_MOTOR_CONFIG["deceleration"])
    print("⚙️ Stepper motor configured.")

def is_ap2_home_position(stepper):
    """
    Checks if the stepper motor is at the home position (GPIO0 LOW).

    Args:
        stepper (BrickletSilentStepperV2): The stepper motor instance.

    Returns:
        bool: True if at home position, False otherwise.
    """
    gpio0_state, _ = stepper.get_gpio_state()
    print(f"GPIO State: {gpio0_state}")

    if gpio0_state is None:
        print("🚨 ERROR: GPIO state is None! Check the connection.")
        return False

    return not gpio0_state  # If False (LOW), return True (Home Position reached)

def ap2_move_stepper(stepper, target_position):
    """
    Moves the stepper motor to the specified target position.

    Args:
        stepper (BrickletSilentStepperV2): The stepper motor instance.
        target_position (int): The target position in steps.
    """
    if target_position > AP2_UPPER_LIMIT or target_position < AP2_LOWER_LIMIT:
        print(f"🚫 ERROR: Position out of range ({AP2_LOWER_LIMIT} to {AP2_UPPER_LIMIT})!")
        return

    stepper.set_target_position(target_position)

    while stepper.get_current_position() != stepper.get_target_position():
        time.sleep(0.1)

    print(f"✅ Movement complete. Current Position: {stepper.get_current_position()} steps.")

def ap2_home_stepper(stepper):
    """
    Moves the stepper motor to home position (GPIO0 LOW) and sets it to zero.

    Args:
        stepper (BrickletSilentStepperV2): The stepper motor instance.
    """
    print("🏠 Starting Homing Process...")
    if stepper.get_current_position() == 0:
        print("Already at home position.")
        return

    stepper.set_enabled(True)
    stepper.set_target_position(0)

    while stepper.get_current_position() != stepper.get_target_position():
        time.sleep(0.1)

    stepper.set_enabled(False)

    if is_ap2_home_position(stepper):
        print("✅ Reached Home Position.")
    else:
        print("❌ Not in home position.")

def ap2_get_valid_position():
    """
    Gets a valid user input for stepper movement within allowed range.

    Returns:
        int: The validated position entered by the user.
    """
    while True:
        try:
            pos = int(input(f"Enter position (must be between {AP2_LOWER_LIMIT} and {AP2_UPPER_LIMIT}): "))
            if AP2_LOWER_LIMIT <= pos <= AP2_UPPER_LIMIT:
                return pos
            else:
                print(f"❌ Invalid range! Must be between {AP2_LOWER_LIMIT} and {AP2_UPPER_LIMIT}.")
        except ValueError:
            print("❌ Invalid input! Please enter a valid integer.")

def ap2_user_menu(stepper):
    """
    Displays the user menu for AirPot-1 movement control.

    Args:
        stepper (BrickletSilentStepperV2): The stepper motor instance.
    """
    while True:
        print("\n🔹 AirPot-1 (2akh) Movement Options:")
        print("1. Move Up (Must be between 0 and -1,970,000)")
        print("2. Move Down (Min: -1,970,000)")
        print("3. Home Position (Detect via GPIO0 LOW)")
        print("4. Exit")

        choice = input("Enter your choice (1/2/3/4): ")
        print(f"Current Position: {stepper.get_current_position()}")

        if choice in ["1", "2"]:
            stepper.set_enabled(True)
            new_position = ap2_get_valid_position()
            ap2_move_stepper(stepper, new_position)
            stepper.set_enabled(False)
        elif choice == "3":
            ap2_home_stepper(stepper)
        elif choice == "4":
            print("🚪 Exiting program.")
            break
        else:
            print("❌ Invalid choice! Please enter 1, 2, 3, or 4.")

def ap2_main():
    """
    Main function to initialize and run the user menu for AirPot-2 movement control.
    """
    ip_con = create_connection()

    stepper = ap2_initialize_stepper(ip_con)

    ap2_user_menu(stepper)

    ip_con.disconnect()

    print("✅ Stepper Control Completed.")

if __name__ == "__main__":
    ap2_main()
